-- Table: Facebook_Posts
CREATE TABLE facebook_posts (
    id BIGINT PRIMARY KEY,
    post_id VARCHAR(255) NOT NULL,
    author_name VARCHAR(255),
    author_id VARCHAR(255),
    content TEXT,
    created_time DATETIME NOT NULL,
    updated_time DATETIME,
    like_count INT DEFAULT 0,
    comment_count INT DEFAULT 0,
    share_count INT DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME ON UPDATE CURRENT_TIMESTAMP
);

-- Table: Facebook_Comments
CREATE TABLE facebook_comments (
    id BIGINT PRIMARY KEY,
    comment_id VARCHAR(255) NOT NULL,
    post_id VARCHAR(255) NOT NULL,
    author_name VARCHAR(255),
    author_id VARCHAR(255),
    parent_comment_id VARCHAR(255),
    content TEXT,
    created_time DATETIME NOT NULL,
    updated_time DATETIME,
    like_count INT DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (post_id) REFERENCES facebook_posts(post_id),
    FOREIGN KEY (parent_comment_id) REFERENCES facebook_comments(comment_id)
);

-- Table: Facebook_Users
CREATE TABLE facebook_users (
    id BIGINT PRIMARY KEY,
    user_id VARCHAR(255) UNIQUE NOT NULL,
    name VARCHAR(255),
    profile_url VARCHAR(255),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME ON UPDATE CURRENT_TIMESTAMP
);

-- Table: Reactions
CREATE TABLE reactions (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    comment_id VARCHAR(255),
    post_id VARCHAR(255),
    user_id VARCHAR(255),
    reaction_type VARCHAR(50),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (post_id) REFERENCES facebook_posts(post_id),
    FOREIGN KEY (comment_id) REFERENCES facebook_comments(comment_id),
    FOREIGN KEY (user_id) REFERENCES facebook_users(user_id)
);

-- Table: Facebook_Fetch_History
CREATE TABLE facebook_fetch_history (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    fetch_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    post_id VARCHAR(255),
    comments_fetched INT DEFAULT 0,
    status VARCHAR(50),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (post_id) REFERENCES facebook_posts(post_id)
);
